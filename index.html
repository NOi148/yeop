/* ===================================================================== style.css ― 커스텀 스타일 & 애니메이션 키프레임 --------------------------------------------------------------------- [CSS 주석 요구사항] - Tailwind 가 기본 다크 테마/레이아웃을 담당하고, 이 파일은 "전자기/열/에너지" 컨셉을 표현하는 네온 글로우와 SVG 애니메이션(피스톤 진동, 빨강 플래시) 전용 키프레임을 정의한다. - 색상 토큰 · 시안(#22d3ee) → 전자기(EMF/코일) · 앰버(#fbbf24) → 발열(Joule Heating) · 에메랄드(#34d399) → 에너지 저장(나노버퍼/HESS) ===================================================================== */ /* 전역 배경: Tailwind 의 bg-slate-900 와 동일 */ body { background:#0f172a; } /* ----- 네온 글로우 유틸리티 ----- */ /* text-shadow 로 살짝 발광하는 듯한 느낌을 줘서 "전자기/열/에너지" 강조 */ .neon-cyan { color:#22d3ee; text-shadow:0 0 6px #22d3ee88; } .neon-amber { color:#fbbf24; text-shadow:0 0 6px #fbbf2488; } .neon-emerald { color:#34d399; text-shadow:0 0 6px #34d39988; } /* ----- 서스펜션 피스톤 상하 진동 ----- */ /* translateY(0 → 28px → 0) 사이클로 노면 충격을 받아 피스톤이 눌렸다가 복귀하는 모습 */ @keyframes piston { 0%, 100% { transform: translateY(0); } 50% { transform: translateY(28px); } } .piston-anim { animation: piston 1.2s ease-in-out infinite; } /* ----- 빨강 플래시 ----- */ /* Conventional 모드 + Speed Bump 일 때 이온 도트가 깜빡이며 "Joule 발열로 인한 열 손실"을 시각적으로 경고 */ @keyframes flash { 0%, 100% { opacity: .2; } 50% { opacity: 1; } } .flash-red { animation: flash .35s ease-in-out infinite; }
