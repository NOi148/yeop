"""
database.py
-----------
SQLite 연결 및 시뮬레이션 로그 저장/조회를 담당하는 모듈.
"""
import sqlite3
from pathlib import Path

# DB 파일 경로 (app.py 와 같은 폴더에 simulation.db 생성)
DB_PATH = Path(__file__).parent / "simulation.db"


def get_conn():
    """SQLite 커넥션을 반환. row_factory 를 dict 형태로 설정."""
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row  # 결과를 dict 처럼 사용 가능
    return conn


def init_db():
    """
    [주석 요구사항 ②] SQLite 스키마 초기화 부분.
    - 서버 기동 시 1회 실행.
    - simulation_history 테이블이 없으면 생성한다.
    - 컬럼: id, timestamp, mode(모드), v(속도), N(코일 턴수),
            R(저항), voltage(유도전압), loss(줄열손실), efficiency(효율)
    """
    with get_conn() as conn:
        conn.execute(
            """
            CREATE TABLE IF NOT EXISTS simulation_history (
                id        INTEGER PRIMARY KEY AUTOINCREMENT,
                timestamp TEXT    DEFAULT (datetime('now','localtime')),
                mode      TEXT    NOT NULL,
                v         REAL    NOT NULL,
                N         INTEGER NOT NULL,
                R         REAL    NOT NULL,
                voltage   REAL    NOT NULL,
                loss      REAL    NOT NULL,
                efficiency REAL   NOT NULL
            )
            """
        )
        conn.commit()


def insert_log(mode, v, N, R, voltage, loss, efficiency):
    """
    [주석 요구사항 ②] INSERT INTO 로 사용자의 실시간 시뮬레이션 선택값을 기록.
    - /api/simulate 가 호출될 때마다 자동으로 한 행이 쌓인다.
    - 파라미터화 쿼리(?)를 사용해 SQL Injection 을 방지한다.
    """
    with get_conn() as conn:
        conn.execute(
            """INSERT INTO simulation_history
               (mode, v, N, R, voltage, loss, efficiency)
               VALUES (?, ?, ?, ?, ?, ?, ?)""",
            (mode, v, N, R, voltage, loss, efficiency),
        )
        conn.commit()


def fetch_recent(limit=5):
    """최근 N개의 시뮬레이션 기록을 최신순으로 반환."""
    with get_conn() as conn:
        rows = conn.execute(
            "SELECT * FROM simulation_history ORDER BY id DESC LIMIT ?",
            (limit,),
        ).fetchall()
        return [dict(r) for r in rows]
